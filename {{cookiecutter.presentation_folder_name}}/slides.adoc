== Go code with callout

// You can use callouts to give context to certain lines of code

[.stretch]
[source,go]
----
package main

import (
	"fyne.io/fyne/app"
	"fyne.io/fyne/widget"
)

func main() {
	a := app.New() // <1>
	w := a.NewWindow("Hello")
	w.SetContent(widget.NewVBox(
		widget.NewLabel("Hello Fyne!"),
		widget.NewButton("Quit", func() { // <2>
			a.Quit()
		}),
	))
	w.ShowAndRun() // <3>
}
----
